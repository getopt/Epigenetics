# Function to run a single experiment
run_experiment <- function(size, ns_mean, ns_sd, cs_mean, cs_sd)
{
    ns_data <- rnorm(size, mean = ns_mean, sd = ns_sd)
    cs_data <- rnorm(size, mean = cs_mean, sd = cs_sd)

    test <- t.test(ns_data, cs_data)
    return(test$p.value)
}
